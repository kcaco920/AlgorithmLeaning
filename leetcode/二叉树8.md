class Solution:
    def traversal(self, nums: List[int], left: int, right: int) -> TreeNode:
        if left > right:
            return None
        
        mid = left + (right - left) // 2
        root = TreeNode(nums[mid])
        root.left = self.traversal(nums, left, mid - 1)
        root.right = self.traversal(nums, mid + 1, right)

        stack = []
        cur = root
        while cur or stack:

                cur = cur.right  # 右
            else:
                cur = stack.pop()  # 中

                self.pre = cur.val
